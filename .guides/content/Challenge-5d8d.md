$$ \begin{align}
	   y &=  3x^{-1} \\
	   &  \\
	   \frac{dy}{dx} &=  3(-1x^{-2}) \\
	    &= -3x^{-2}  \\
	    &= -\frac{3}{x^{2}}  \\
	    & \\
	    \frac{d^{2}y}{dx^{2}}  &= -3(-2x^{-3}) \\
	    &= \frac{6}{x^{3}}   
	    \end{align}      
$$
